if (NOT TARGET test-afsm-base)

    find_package(Threads REQUIRED)

    simple_test.cpp

    test-afsm-base
    ${GTEST_BOTH_LIBRARIES}
    ${CMAKE_THREAD_LIBS_INIT}
)

        -Wno-rtti -Wno-exceptions
